import { useState } from 'react';

import { User } from '../../App';
import { BookVehicle } from './BookVehicle';
import ZomatoBookVehicle from './ZomatoBookVehicle';
import { MyBookings } from './MyBookings';
import { SavedAddresses } from './SavedAddresses';
import { Profile } from '../shared/Profile';
import { EmergencyCall } from '../shared/EmergencyCall';

interface CustomerDashboardProps {

  onLogout: () => void;
}

export function CustomerDashboard({ user, onLogout }: CustomerDashboardProps) {
  const [activeTab, setActiveTab] = useState<'home' | 'book' | 'bookings' | 'addresses' | 'profile'>('home');

  const [showEmergency, setShowEmergency] = useState(false);

            >

            </button>
          </nav>
        </div>

        {/* Mobile Navigation */}
        {menuOpen && (
          <nav className="lg:hidden border-t border-gray-200 py-2">
            <button
              onClick={() => { setActiveTab('home'); setMenuOpen(false); }}
              className={`w-full px-4 py-3 text-left ${activeTab === 'home' ? 'bg-blue-50 text-blue-600' : 'text-gray-600'}`}
            >
              <Home className="w-5 h-5 inline mr-2" />

            >
              <Package className="w-5 h-5 inline mr-2" />
              Book Vehicle
            </button>
            <button
              onClick={() => { setActiveTab('bookings'); setMenuOpen(false); }}
              className={`w-full px-4 py-3 text-left ${activeTab === 'bookings' ? 'bg-blue-50 text-blue-600' : 'text-gray-600'}`}
            >
              <Clock className="w-5 h-5 inline mr-2" />
              My Bookings
            </button>
            <button
              onClick={() => { setActiveTab('addresses'); setMenuOpen(false); }}
              className={`w-full px-4 py-3 text-left ${activeTab === 'addresses' ? 'bg-blue-50 text-blue-600' : 'text-gray-600'}`}

            >
              <UserIcon className="w-5 h-5 inline mr-2" />
              Profile
            </button>
          </nav>
        )}
      </header>

      {/* Main Content */}
      {activeTab === 'book' && modernUI ? (
        // Zomato-style Premium UI (full screen)
        <ZomatoBookVehicle user={user} />
      ) : (
        <main className="max-w-7xl mx-auto p-4">
          {activeTab === 'home' && <HomeTab user={user} onBookNow={() => setActiveTab('book')} />}
          {activeTab === 'book' && (
            <div>
              {/* Toggle Button */}
              <div className="mb-4 flex justify-end">
                <button
                  onClick={() => setModernUI(!modernUI)}
                  className="flex items-center gap-2 px-4 py-2 bg-gradient-to-r from-red-500 to-orange-500 text-white rounded-lg hover:shadow-lg transition-all font-semibold"
                >
                  <Sparkles className="w-5 h-5" />
                  Switch to {modernUI ? 'Classic' : 'Premium'} View
                </button>
              </div>
              <BookVehicle user={user} />
            </div>
          )}
          {activeTab === 'bookings' && <MyBookings user={user} />}
          {activeTab === 'addresses' && <SavedAddresses user={user} />}
          {activeTab === 'profile' && <Profile user={user} onLogout={onLogout} />}
        </main>
      )}

      {/* Emergency Call Modal */}
      {showEmergency && (
        <EmergencyCall
          onClose={() => setShowEmergency(false)}
          adminPhone="+91 1800-123-4567"
        />
      )}
    </div>
  );
}
